"""
Frigate tools — access camera events, recordings, stats via the Frigate REST API.
"""

import httpx
from typing import Optional, List


def register_frigate_tools(mcp, ctx: dict):

    frigate_url = ctx["frigate_url"]

    def _get(path: str, params: dict = None) -> dict:
        try:
            r = httpx.get(f"{frigate_url}{path}", params=params or {}, timeout=10)
            r.raise_for_status()
            return r.json()
        except Exception as e:
            return {"error": str(e)}

    @mcp.tool()
    def frigate_health() -> dict:
        """Check Frigate connectivity and version."""
        try:
            r = httpx.get(f"{frigate_url}/api/version", timeout=5)
            return {"connected": True, "version": r.text.strip()}
        except Exception as e:
            return {"connected": False, "error": str(e)}

    @mcp.tool()
    def frigate_get_cameras() -> dict:
        """
        List all configured Frigate cameras with their status and detection config.
        """
        data = _get("/api/config")
        if "error" in data:
            return data
        cameras = data.get("cameras", {})
        result = {}
        for name, config in cameras.items():
            result[name] = {
                "enabled": config.get("enabled", True),
                "detect": config.get("detect", {}).get("enabled", True),
                "objects": config.get("objects", {}).get("track", []),
                "record": config.get("record", {}).get("enabled", False),
                "snapshot": config.get("snapshots", {}).get("enabled", False),
            }
        return {"cameras": result, "count": len(result)}

    @mcp.tool()
    def frigate_get_events(
        camera: Optional[str] = None,
        label: Optional[str] = None,
        limit: int = 20,
        has_clip: Optional[bool] = None,
        has_snapshot: Optional[bool] = None,
        after: Optional[int] = None,
        before: Optional[int] = None,
    ) -> dict:
        """
        Get Frigate detection events (person, car, dog, etc.).
        
        Args:
            camera: Filter by camera name (e.g. 'oprit', 'voordeur').
            label: Filter by object label (e.g. 'person', 'car', 'dog').
            limit: Max events to return (default 20).
            has_clip: Only events with/without video clip.
            has_snapshot: Only events with/without snapshot.
            after: Unix timestamp — only events after this time.
            before: Unix timestamp — only events before this time.
        """
        params = {"limit": limit}
        if camera:      params["camera"] = camera
        if label:       params["label"] = label
        if has_clip is not None:     params["has_clip"] = int(has_clip)
        if has_snapshot is not None: params["has_snapshot"] = int(has_snapshot)
        if after:       params["after"] = after
        if before:      params["before"] = before

        data = _get("/api/events", params)
        if "error" in data:
            return data

        events = data if isinstance(data, list) else data.get("events", [])
        return {
            "count": len(events),
            "events": [
                {
                    "id": e.get("id"),
                    "camera": e.get("camera"),
                    "label": e.get("label"),
                    "score": round(e.get("score", 0) * 100, 1),
                    "start_time": e.get("start_time"),
                    "end_time": e.get("end_time"),
                    "has_clip": e.get("has_clip"),
                    "has_snapshot": e.get("has_snapshot"),
                    "snapshot_url": f"{frigate_url}/api/events/{e.get('id')}/snapshot.jpg" if e.get("has_snapshot") else None,
                    "clip_url": f"{frigate_url}/api/events/{e.get('id')}/clip.mp4" if e.get("has_clip") else None,
                }
                for e in events
            ],
        }

    @mcp.tool()
    def frigate_get_stats() -> dict:
        """
        Get Frigate system statistics: detector performance, CPU, GPU, camera FPS.
        """
        return _get("/api/stats")

    @mcp.tool()
    def frigate_get_event_counts(
        camera: Optional[str] = None,
        label: Optional[str] = None,
        hours: int = 24,
    ) -> dict:
        """
        Get event counts grouped by camera and label for the last N hours.
        Useful for trend analysis and Grafana panels.
        
        Args:
            camera: Filter by camera name.
            label: Filter by object label.
            hours: Look back this many hours (default 24).
        """
        import time as _time
        after = int(_time.time()) - hours * 3600
        params = {"after": after, "limit": 1000}
        if camera: params["camera"] = camera
        if label:  params["label"] = label

        data = _get("/api/events", params)
        if "error" in data:
            return data
        events = data if isinstance(data, list) else data.get("events", [])

        counts: dict = {}
        for e in events:
            key = f"{e.get('camera')} / {e.get('label')}"
            counts[key] = counts.get(key, 0) + 1

        return {
            "period_hours": hours,
            "total_events": len(events),
            "by_camera_label": dict(sorted(counts.items(), key=lambda x: x[1], reverse=True)),
        }

    @mcp.tool()
    def frigate_get_recordings(camera: str, date: Optional[str] = None) -> dict:
        """
        Get recording segments for a camera on a specific date.
        
        Args:
            camera: Camera name (e.g. 'oprit').
            date: Date string in YYYY-MM-DD format. Defaults to today.
        """
        from datetime import date as _date
        d = date or _date.today().isoformat()
        return _get(f"/api/{camera}/recordings", {"before": d + "T23:59:59", "after": d + "T00:00:00"})

    @mcp.tool()
    def frigate_get_labels() -> dict:
        """List all object labels Frigate is configured to detect."""
        data = _get("/api/labels")
        return {"labels": data} if isinstance(data, list) else data
